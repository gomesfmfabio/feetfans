🔧 ALTERNATIVA - COPIAR E COLAR NO CONSOLE:

Se bookmarklet não funcionar, copie o código abaixo, cole no Console (F12) do browser estando logado no N8N:

// ═══════════════════════════════════════════════════════════════
// 🚀 N8N AUTO-DEPLOY VIA CONSOLE
// ═══════════════════════════════════════════════════════════════

console.log('🚀 Iniciando deploy automatizado...');

// Workflow completo (base64 encoded)
const workflowBase64 = 'WORKFLOW_BASE64_PLACEHOLDER';

// Decode
const workflowJSON = JSON.parse(atob(workflowBase64));

console.log('📝 Workflow:', workflowJSON.name);
console.log('📊 Nodes:', workflowJSON.nodes.length);

// Import via API
fetch('/rest/workflows', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json'
  },
  body: JSON.stringify(workflowJSON),
  credentials: 'include'
})
.then(response => response.json())
.then(data => {
  console.log('✅ Workflow importado!');
  console.log('   ID:', data.id);
  console.log('   Nome:', data.name);

  // Redirecionar para o workflow
  alert('✅ Deploy completo! Redirecionando...');
  window.location.href = '/workflow/' + data.id;
})
.catch(error => {
  console.error('❌ Erro:', error);
  alert('❌ Erro ao importar workflow. Ver console para detalhes.');
});
