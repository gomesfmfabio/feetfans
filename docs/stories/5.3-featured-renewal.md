// Check featured status
const { data: featured } = await supabase
  .from('featured_placements')
  .select('expires_at, tier')
  .eq('creator_id', userId)
  .gt('expires_at', new Date())
  .single();

const daysRemaining = featured ? daysBetween(new Date(), featured.expires_at) : 0;

// Auto-renew for recurring (Stripe webhook)
if (event.type === 'invoice.payment_succeeded' && subscription.metadata.type === 'featured') {
  await supabase
    .from('featured_placements')
    .update({ expires_at: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000) })
    .eq('creator_id', subscription.metadata.creator_id);
}
