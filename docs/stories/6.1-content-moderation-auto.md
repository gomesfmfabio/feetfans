## Technical Notes

```typescript
// Sightengine integration
import axios from 'axios';

async function moderateImage(imageUrl: string) {
  const response = await axios.get('https://api.sightengine.com/1.0/check.json', {
    params: {
      url: imageUrl,
      models: 'nudity,violence,offensive',
      api_user: process.env.SIGHTENGINE_API_USER,
      api_secret: process.env.SIGHTENGINE_API_SECRET,
    },
  });

  const { nudity, violence, offensive } = response.data;

  // Reject if explicit content detected
  if (nudity.sexual_activity > 0.8 || violence.prob > 0.8 || offensive.prob > 0.8) {
    return { approved: false, reason: 'Violates content policy' };
  }

  return { approved: true, labels: response.data };
}
```
