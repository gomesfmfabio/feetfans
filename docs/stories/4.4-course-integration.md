export async function POST(request: Request) {
  const { course_id, email } = await request.json();

  // Verify with database or Hotmart API
  const isValid = await verifyCourseGraduate(email, course_id);

  if (isValid) {
    await supabase
      .from('users')
      .update({
        subscription_status: 'free_feetfans',
        trial_ends_at: null,
        feetfans_course_id: course_id,
      })
      .eq('email', email);

    return Response.json({ success: true });
  }

  return Response.json({ error: 'Invalid course ID' }, { status: 400 });
}
