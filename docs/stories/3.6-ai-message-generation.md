## Technical Notes

```typescript
import Anthropic from '@anthropic-ai/sdk';

const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

async function generateAgentMessage(agentId: string, creatorId: string) {
  // Fetch agent personality
  const { data: agent } = await supabase
    .from('ai_agents')
    .select('personality_prompt, nickname')
    .eq('id', agentId)
    .single();

  // Fetch creator's recent content
  const { data: content } = await supabase
    .from('content')
    .select('categories')
    .eq('creator_id', creatorId)
    .order('created_at', { ascending: false })
    .limit(5);

  const categories = content.flatMap(c => c.categories).join(', ');

  // Construct prompt
  const prompt = `${agent.personality_prompt}\n\nThe creator has posted content in these categories: ${categories}.\n\nWrite a brief, friendly opening message (max 50 words) expressing interest in their content. Do not reveal you are AI.`;

  // Call Claude Haiku
  const message = await anthropic.messages.create({
    model: 'claude-3-haiku-20240307',
    max_tokens: 100,
    messages: [{ role: 'user', content: prompt }],
  });

  // Log cost
  console.log(`Tokens used: ${message.usage.input_tokens + message.usage.output_tokens}`);

  return message.content[0].text;
}
```
