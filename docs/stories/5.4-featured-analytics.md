## Technical Notes

```sql
CREATE TABLE analytics_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  event_type TEXT NOT NULL,
  metadata JSONB,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_analytics_user_type ON analytics_events(user_id, event_type);
CREATE INDEX idx_analytics_created_at ON analytics_events(created_at);
```

```typescript
// Track event
await supabase.from('analytics_events').insert({
  user_id: creatorId,
  event_type: 'profile_view',
  metadata: { viewer_id: viewerId, source: 'feed' },
});

// Aggregate stats
const { count: viewsWhileFeatured } = await supabase
  .from('analytics_events')
  .select('*', { count: 'exact', head: true })
  .eq('user_id', creatorId)
  .eq('event_type', 'profile_view')
  .gte('created_at', featuredStart)
  .lte('created_at', featuredEnd);
```
