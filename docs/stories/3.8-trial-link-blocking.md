## Technical Notes

```typescript
const URL_REGEX = /https?:\/\/[^\s]+/g;

function stripLinks(message: string, userStatus: string): string {
  if (userStatus === 'trial') {
    return message.replace(URL_REGEX, '[link removed]');
  }
  return message;
}

// Before passing to AI
const processedMessage = stripLinks(creatorMessage, creator.subscription_status);
```
