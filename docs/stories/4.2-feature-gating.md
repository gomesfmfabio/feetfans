export async function checkFeatureAccess(userId: string, feature: string): Promise<boolean> {
  const { data: user } = await supabase
    .from('users')
    .select('subscription_status')
    .eq('id', userId)
    .single();

  if (feature === 'message_real_users') {
    return ['paid', 'free_feetfans'].includes(user.subscription_status);
  }

  return true;
}

## Dev Agent Record

### Implementation Summary
Created comprehensive feature gating system with backend enforcement and UI feedback for trial vs paid users.

### Technical Decisions
- **Feature-based permissions:** Granular features (upload, message_ai, message_real, etc.)
- **Backend enforcement:** API endpoint checks before allowing message send
- **AI agent detection:** Check if recipient is in ai_agents table
- **Subscription status types:** trial, active, free_feetfans, expired, cancelled, consumer
- **Upgrade banner:** Prominent red banner with CTA when trial user tries to message real buyer

### Features Implemented
**Feature Gating Middleware:**
- checkFeatureAccess(userId, feature) - Main permission check
- getSubscriptionStatus(userId) - Get user's current status
- isPaidUser(userId) - Check if user is paid (active or free_feetfans)
- isTrialUser(userId) - Check if user is on trial
- isExpiredUser(userId) - Check if subscription expired

**Feature Access Matrix:**
- upload_content: trial + paid creators
- view_feed/discovery: all users
- message_ai_agents: trial + paid creators
- message_real_users: ONLY paid creators
- receive_messages: trial + paid creators (expired = read-only)
- featured_placement: ONLY paid creators

**Backend Enforcement (POST /api/messages):**
- Determine recipient from conversation
- Check if recipient is AI agent (ai_agents table lookup)
- If real user, verify sender has message_real_users permission
- Return 403 with upgrade_required flag if blocked

**UI Feedback:**
- Detect AI agent vs real user (isAIAgent state)
- Show upgrade banner when message blocked (upgradeRequired state)
- Banner shows: "Upgrade to Message Real Buyers" with CTA
- Existing trial banner shows for trial users
- Graceful error handling with input restoration
